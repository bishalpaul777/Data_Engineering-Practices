|| Enter some data in a file locally ||

echo 1, Aditya, 25 > manfile
echo 2, deb, 28 >> manfile



|| Create external table ||

hive > create external table exttab( id int, name string, age int) row format delimited fields terminated by ',';


|| load the data ||

hive > load data local inpath '/home/cloudera/manfile' into table exttab;



|| This will create a directory in the name of table in the hive warehouse ||

Hadoop fs -ls /user/hive/warehouse    <---type
>> exttab     <-- directory
Hadoop fs -ls /user/hive/warehouse/exttab       <----type
>> manfile





|| If we delete the hive table, it will not delete the directory ||

hive > drop table exttab;




|| The back end file will remain same because it's a external table ||
